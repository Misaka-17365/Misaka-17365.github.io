STM32F401CCU6 使用手册

EXTI 外部中断

STM32F401CCU6 共有 24 个外部中断，其中中断 0 ~ 15 可由 GPIO 配置，中断 16 ~ 23 连接到特定的外设上，实现外设的一些特殊功能。

这里主要列出通过 GPIO 来配置外部中断的方式。

开启外部中断，需要开启的时钟：


// 开启使用的 GPIO 的时钟
RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOx, ENABLE);
// 开启 AF 的时钟
RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);

需要将 GPIO 设定为输入模式：


GPIO_InitTypeDef GPIO_InitStructure;
GPIO_InitStructure.GPIO_Mode    = GPIO_Mode_IN;
GPIO_InitStructure.GPIO_OType   = GPIO_OType_OD;
GPIO_InitStructure.GPIO_Pin     = GPIO_Pin_x;
GPIO_InitStructure.GPIO_PuPd    = GPIO_PuPd_NOPULL;
GPIO_InitStructure.GPIO_Speed   = GPIO_Speed_50MHz;
GPIO_Init(GPIOx, &GPIO_InitStructure);

EXTI 结构

GPIO Pin 0 ~ 15 通过 AF 选择器输送到外部中断线 EXTI 0 ~ 15 。

外部中断线 EXTI 0 ~ 23 进入中断控制。

由图右下角输入。

更正：图中线上‘20’应改为‘24’，表示24根线。

EXTI 的路径为：GPIO引脚 -> AFIO 配置 -> EXTI 配置 -> NVIC 中断优先级控制 -> 内核执行

由于多个端口的同名引脚连接到了同一个选择器上，因此配置外部中断的时候，同一个 Pin 只能使用一个。

触发方式可以分为：上升沿触发，下降沿触发，双边缘触发，事件触发。

外部触发可以选择触发中断和触发事件。

中断优先级有两类: 抢占优先级和响应优先级。

抢占优先级：优先级高的可以打断当前正在进行的任务，执行中断。

响应优先级：优先级高的可以优先被处理（不能抢占任务）。

优先级配置中，数值越小，优先级越大。

中断函数的函数名是固定的，函数名在startup_stm32f401xx.s文件里。

EXTI 基本操作流程

将 GPIOC-14 作为外部触发源。


int main(void)
{
    GPIO_InitTypeDef GPIO_InitStructure;
    EXTI_InitTypeDef EXTI_InitStructure;
    NVIC_InitTypeDef NVIC_InitStructure;

    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOC, ENABLE);
    RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);

    SYSCFG_EXTILineConfig(EXTI_PortSourceGPIOC, GPIO_PinSource14);

    GPIO_InitStructure.GPIO_Mode    = GPIO_Mode_IN;
    GPIO_InitStructure.GPIO_OType   = GPIO_OType_OD;
    GPIO_InitStructure.GPIO_Pin     = GPIO_Pin_14;
    GPIO_InitStructure.GPIO_PuPd    = GPIO_PuPd_UP;
    GPIO_InitStructure.GPIO_Speed   = GPIO_Speed_50MHz;

    GPIO_Init(GPIOC, &GPIO_InitStructure);

    EXTI_InitStructure.EXTI_Line        = EXTI_Line14;
    EXTI_InitStructure.EXTI_LineCmd     = ENABLE;
    EXTI_InitStructure.EXTI_Mode        = EXTI_Mode_Interrupt;
    EXTI_InitStructure.EXTI_Trigger     = EXTI_Trigger_Falling;

    EXTI_Init(&EXTI_InitStructure);

    NVIC_PriorityGroupConfig(NVIC_PriorityGroup_2);
    NVIC_InitStructure.NVIC_IRQChannel                      = EXTI15_10_IRQn;
    NVIC_InitStructure.NVIC_IRQChannelCmd                   = ENABLE;
    NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority    = 1;
    NVIC_InitStructure.NVIC_IRQChannelSubPriority           = 1;

    NVIC_Init(&NVIC_InitStructure);

    while(1);
}

void EXTI15_10_IRQHandler(void)
{
	if(EXTI_GetITStatus(EXTI_Line14) == SET){
		
		EXTI_ClearITPendingBit(EXTI_Line14);
	}
}

AFIO 配置

首先开启时钟。

SYSCFG 在 APB2 总线上。


RCC_APB2PeriphClockCmd(RCC_APB2Periph_SYSCFG, ENABLE);

将 GPIO 的引脚设定为外部触发源。


SYSCFG_EXTILineConfig(uint8_t EXTI_PortSourceGPIOx, uint8_t EXTI_PinSourcex);

EXTI_PortSourceGPIOx外部触发选用的 GPIO 端口

EXTI_PinSourcex外部触发选用的引脚

设定好引脚之后会自动连接到外部中断线上，无需手动设置。

示例：


// 将 GPIOA-10 接到 EXTI 10 上
SYSCFG_EXTILineConfig(EXTI_PortSourceGPIOA, EXTI_PinSource10);

EXTI 配置

配置状态结构体


typedef struct
{
    uint32_t E              XTI_Line;       // 选定的触发线    
    EXTIMode_TypeDef        EXTI_Mode;      // 触发模式
    EXTITrigger_TypeDef     EXTI_Trigger;   // 触发沿
    FunctionalState         EXTI_LineCmd;   // 触发线控制
}EXTI_InitTypeDef;

该结构体用于设定 EXTI 的状态。

EXTI_LineEXTI_Line0 ~ EXTI_Line23

EXTI_ModeEXTI_Mode_Interrupt, EXTI_Mode_Event

EXTI_TriggerEXTI_Trigger_Rising, EXTI_Trigger_Falling, EXTI_Trigger_Rising_Falling

EXTI_LineCmdENABLE, DISABLE

配置 NVIC

状态配置结构体


typedef struct
{
    uint8_t NVIC_IRQChannel;                    // 选择中断请求的线路         
    uint8_t NVIC_IRQChannelPreemptionPriority;  // 设置抢占优先级
    uint8_t NVIC_IRQChannelSubPriority;         // 设置响应优先级
    FunctionalState NVIC_IRQChannelCmd;         // 设置使能
} NVIC_InitTypeDef;

设定 NVIC 的优先级分组


void NVIC_PriorityGroupConfig(uint32_t NVIC_PriorityGroup);

NVIC_PriorityGourpNVIC_PriorityGourp_0, NVIC_PriorityGourp_1, NVIC_PriorityGourp_3, NVIC_PriorityGourp_3, NVIC_PriorityGourp_4


@arg NVIC_PriorityGroup_0:  0 bits for pre-emption priority | 4 bits for subpriority
@arg NVIC_PriorityGroup_1:  1 bits for pre-emption priority | 3 bits for subpriority
@arg NVIC_PriorityGroup_2:  2 bits for pre-emption priority | 2 bits for subpriority
@arg NVIC_PriorityGroup_3:  3 bits for pre-emption priority | 1 bits for subpriority
@arg NVIC_PriorityGroup_4:  4 bits for pre-emption priority | 0 bits for subpriority

示例：


// 将 NVIC 设置为：1位抢占优先级，3位响应优先级
NVIC_PriorityGroupConfig(NVIC_PriorityGourp_1);
