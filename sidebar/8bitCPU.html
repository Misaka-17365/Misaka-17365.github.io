                    return -1;
                }
                int flag = findRx(word[1]);
                if (flag == -1) {
                    cmd[cmdCount]->pa_1_type = 4;
                    strcpy(cmd[cmdCount]->pa_1_value, word[1]);
                }
                else {
                    cmd[cmdCount]->pa_1_type = flag;
                    (cmd[cmdCount]->pa_1_value)[0] = 0;
                }
        
        
                if (word[2][0] == 0) {
                    cmd[cmdCount]->pa_2_type = -1;
                    (cmd[cmdCount]->pa_2_value)[0] = 0;
                }
                else {
                    flag = findRx(word[2]);

                        cmd[cmdCount]->pa_2_type = 4;

                    }

                        cmd[cmdCount]->pa_2_type = flag;
                        (cmd[cmdCount]->pa_2_value)[0] = 0;
                    }
                }
                cmdCount++;
            }
            
            
        
            // 翻译
            fputs("v2.0 raw\n", out);
            for (int q = 0; q < cmdCount; q++) {
                if (translate(out, cmd[q]) == -1) {
                    printf("error in line: %d", q);
                    return -1;
                }
            }
            
            
            for (int i = 0; i < cmdCount; i++) {
                printf("%d\t|\t%d\t:\t%s\t|\t%d\t:\t%s\n", cmd[i]->mne, cmd[i]->pa_1_type, cmd[i]->pa_1_value, cmd[i]->pa_2_type, cmd[i]->pa_2_value);
            }
            
            fclose(in);
            fclose(out);

            return 0;

        int findMne(char* str)
        {
            int i;
            for (i = 0; i < 21; i++) {
                if (0 == strcmp(str, MNEMONIC[i])) {
                    break;
                }
            }
            if (i == 21) {
                i = -1;
            }
            return i;
        }

        int findRx(char* str)
        {
            int i = -1;
            for (i = 0; i < 4; i++) {
                if (0 == strcmp(str, RX[i])) {
                    break;
                }
            }
            if (i == 4) {
                i = -1;
            }
            return i;
        }

        int distruct(char buffer[3][8], char* src)
        {
            for (int i = 0; i < 8; i++) {
                buffer[0][i] = buffer[1][i] = buffer[2][i] = 0;
            }

            int i = 0;
            int wc = 0, j = 0;

            while (src[i] != 0 && src[i] != ';') {
                
                if (src[i] == ' ' || src[i] == '\n' || src[i] == ',' || src[i] == '\t') {
                    while (src[i] == ' ' || src[i] == '\n' || src[i] == ',' || src[i] == '\t') {
                        j = 0;
                        i++;
                    }
                }

                else if (src[i] >= 'A' && src[i] <= 'Z' || src[i] >= '0' && src[i] <= '9') {
                    while (src[i] >= 'A' && src[i] <= 'Z' || src[i] >= '0' && src[i] <= '9') {
                        buffer[wc][j] = src[i];
                        i++;
                        j++;
                    }
                    buffer[wc][j] = 0;
                    wc++;
                }

                else {
                    return -1;
                }
            }
            return 0;
        }

        unsigned char toHex(char ch)
        {
            if (ch >= '0' && ch <= '9') {
                return (ch - '0');
            }
            if (ch >= 'a' && ch <= 'f') {
                return(ch - 'a' + 10);
            }
            if (ch >= 'A' && ch <= 'F') {
                return (ch - 'A' + 10);
            }
        }
